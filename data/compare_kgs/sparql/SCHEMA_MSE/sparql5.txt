# What are the organizations present in the MSE-KG?

PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX nfdi: <https://nfdi.fiz-karlsruhe.de/ontology/>
PREFIX mwo: <http://purls.helmholtz-metadaten.de/mwo/>
PREFIX schema: <http://schema.org/>
PREFIX dct: <http://purl.org/dc/terms/>
PREFIX swo: <http://www.ebi.ac.uk/swo/>
PREFIX foaf: <http://xmlns.com/foaf/0.1/>
SELECT ?org ?label_en ?acronym ?city ?rorID WHERE {
  ?org a schema:Organization .
  OPTIONAL { ?org rdfs:label ?label_en . }
  OPTIONAL { ?org mwo:hasAcronym ?acronym . }
  OPTIONAL { ?org nfdi:location ?c . ?c a nfdi:City . ?c rdfs:label ?city . }
  OPTIONAL { ?org mwo:hasRORID ?rorID . }
}
LIMIT 999
