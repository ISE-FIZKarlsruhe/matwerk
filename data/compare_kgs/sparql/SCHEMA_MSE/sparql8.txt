# What are the data portals present in the MSE-KG?

PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX nfdi: <https://nfdi.fiz-karlsruhe.de/ontology/>
PREFIX mwo: <http://purls.helmholtz-metadaten.de/mwo/>
PREFIX schema: <http://schema.org/>
PREFIX dct: <http://purl.org/dc/terms/>
PREFIX swo: <http://www.ebi.ac.uk/swo/>
PREFIX foaf: <http://xmlns.com/foaf/0.1/>
SELECT ?portal ?name ?link ?repository ?contactpointName ?Email ?Website WHERE {
  ?portal a nfdi:DataPortal .
  OPTIONAL { ?portal dct:title ?name . }
  OPTIONAL { ?portal nfdi:url ?link . }
  OPTIONAL { ?portal mwo:hasRepository ?repositoryNode . OPTIONAL { ?repositoryNode nfdi:url ?repository . } }
  OPTIONAL {
    ?portal mwo:hasContactPoint ?cp .
    OPTIONAL { ?cp rdfs:label ?contactpointName . }
    OPTIONAL { ?cp mwo:emailAddress ?Email . }
    OPTIONAL { ?cp mwo:hasWebsite ?Website . }
  }
}
LIMIT 999
