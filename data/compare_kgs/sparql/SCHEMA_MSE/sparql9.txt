# What are the instruments present in the MSE-KG?

PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX nfdi: <https://nfdi.fiz-karlsruhe.de/ontology/>
PREFIX mwo: <http://purls.helmholtz-metadaten.de/mwo/>
PREFIX schema: <http://schema.org/>
PREFIX dct: <http://purl.org/dc/terms/>
PREFIX swo: <http://www.ebi.ac.uk/swo/>
PREFIX foaf: <http://xmlns.com/foaf/0.1/>
SELECT DISTINCT ?instrument ?name ?contactpointName ?Email ?Website WHERE {
  ?instrument a <https://w3id.org/pmd/co/PMD_0000602> .
  OPTIONAL { ?instrument rdfs:label ?name . }
  OPTIONAL {
    ?instrument mwo:hasContactPoint ?cp .
    OPTIONAL { ?cp rdfs:label ?contactpointName . }
    OPTIONAL { ?cp mwo:emailAddress ?Email . }
    OPTIONAL { ?cp mwo:hasWebsite ?Website . }
  }
}
