# What are the services present in the MSE-KG?

PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX nfdi: <https://nfdi.fiz-karlsruhe.de/ontology/>
PREFIX mwo: <http://purls.helmholtz-metadaten.de/mwo/>
PREFIX schema: <http://schema.org/>
PREFIX dct: <http://purl.org/dc/terms/>
PREFIX swo: <http://www.ebi.ac.uk/swo/>
PREFIX foaf: <http://xmlns.com/foaf/0.1/>
SELECT DISTINCT ?service ?serviceLabel ?serviceURL ?docURL ?codeURL WHERE {
  ?service a schema:Service .
  OPTIONAL { ?service rdfs:label ?serviceLabel . }
  OPTIONAL { ?service nfdi:url ?serviceURL . }
  OPTIONAL { ?service mwo:hasDocumentation ?docURL . }
  OPTIONAL { ?service mwo:hasRepository ?repo . OPTIONAL { ?repo nfdi:url ?codeURL } }
}
LIMIT 999
