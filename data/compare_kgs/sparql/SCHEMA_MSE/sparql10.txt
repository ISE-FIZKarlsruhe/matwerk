# What are the larg scale facilities present in the MSE-KG?

PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX nfdi: <https://nfdi.fiz-karlsruhe.de/ontology/>
PREFIX mwo: <http://purls.helmholtz-metadaten.de/mwo/>
PREFIX schema: <http://schema.org/>
PREFIX dct: <http://purl.org/dc/terms/>
PREFIX swo: <http://www.ebi.ac.uk/swo/>
PREFIX foaf: <http://xmlns.com/foaf/0.1/>
SELECT ?facility ?name ?acronym ?orgLabel ?Email ?Website WHERE {
  ?facility a mwo:MWO_0001027 .
  OPTIONAL { ?facility rdfs:label ?name . }
  OPTIONAL { ?facility mwo:hasAcronym ?acronym . }
  OPTIONAL { ?facility mwo:hasAffiliation ?org . OPTIONAL { ?org rdfs:label ?orgLabel . } }
  OPTIONAL {
     ?facility mwo:hasContactPoint ?cp .
     OPTIONAL { ?cp mwo:emailAddress ?Email . }
     OPTIONAL { ?cp mwo:hasWebsite ?Website . }
  }
}
LIMIT 999
